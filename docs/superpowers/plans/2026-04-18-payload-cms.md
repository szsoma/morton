# Payload CMS Integration — Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Integrate Payload CMS into the Morton Backpacks Next.js app so all section content is editable via an admin panel.

**Architecture:** Payload embedded in the same Next.js 16 app using SQLite for storage. Section content stored as Payload globals, products as a collection. Data fetched via Local API in server components — no HTTP overhead.

**Tech Stack:** Payload 3.x, @payloadcms/db-sqlite, @payloadcms/richtext-lexical, @payloadcms/next, Next.js 16, TypeScript

export const Products: CollectionConfig = {
  slug: "products",

    useAsTitle: "name",
    defaultColumns: ["name", "price", "isNew", "order"],
  },
  fields: [
    {
      name: "name",
      type: "text",
      required: true,
    },
    {
      name: "slug",
      type: "text",
      unique: true,
      index: true,
      admin: {
        readOnly: true,
        hidden: true,
      },
    },
    {
      name: "price",
      type: "number",
      required: true,
    },
    {
      name: "image",
      type: "upload",
      relationTo: "media",
      required: true,
    },
    {
      name: "isNew",
      type: "checkbox",
      defaultValue: false,
    },
    {
      name: "order",
      type: "number",
      defaultValue: 0,
    },
  ],
  hooks: {
    beforeChange: [
      ({ data, operation }) => {
        if (operation === "create" || operation === "update") {
          if (data.name) {
            data.slug = data.name
              .toLowerCase()
              .replace(/[^a-z0-9]+/g, "-")
              .replace(/(^-|-$)/g, "");
          }
        }
        return data;
      },
    ],
  },
};
